101def clear(what: str = "screen") -> None:
102    """Clears the specified screen region.
103
104    Args:
105        what: The specifier defining the screen area.
106
107    Available options:
108    * screen: clear whole screen and go to origin
109    * bos: clear screen from cursor backwards
110    * eos: clear screen from cursor forwards
111    * line: clear line and go to beginning
112    * bol: clear line from cursor backwards
113    * eol: clear line from cursor forwards
114    """
115
116    commands = {
117        "eos": "\x1b[0J",
118        "bos": "\x1b[1J",
119        "screen": "\x1b[2J",
120        "eol": "\x1b[0K",
121        "bol": "\x1b[1K",
122        "line": "\x1b[2K",
123    }
124
125    get_terminal().write(commands[what])

156def report_cursor() -> "Optional[tuple[int, int]]":
157    """Gets position of cursor.
158
159    Returns:
160        A tuple of integers, (columns, rows), describing the
161        current (printing) cursor's position. Returns None if
162        this could not be determined.
163
164        Note that this position is **not** the mouse position. See
165        `report_mouse` if that is what you are interested in.
166    """
167
168    get_terminal().write("\x1b[6n", flush=True)
169    chars = getch()
170    posy, posx = chars[2:-1].split(";")
171
172    if not posx.isdigit() or not posy.isdigit():
173        return None
174
175    return int(posx), int(posy)

301def set_mode(mode: Union[str, int], write: bool = True) -> str:
302    """Sets terminal display mode.
303
304    This is better left internal. To use these modes, you can call their
305    specific functions, such as `bold("text")` or `italic("text")`.
306
307    Args:
308        mode: One of the available modes. Strings and integers both work.
309        write: Boolean that determines whether the output should be written
310            to stdout.
311
312    Returns:
313        A string that sets the given mode.
314
315    Available modes:
316        - 0: reset
317        - 1: bold
318        - 2: dim
319        - 3: italic
320        - 4: underline
321        - 5: blink
322        - 7: inverse
323        - 8: invisible
324        - 9: strikethrough
325        - 53: overline
326    """
327
328    options = {
329        "reset": 0,
330        "bold": 1,
331        "dim": 2,
332        "italic": 3,
333        "underline": 4,
334        "blink": 5,
335        "inverse": 7,
336        "invisible": 8,
337        "strikethrough": 9,
338        "overline": 53,
339    }
340
341    if not str(mode).isdigit():
342        mode = options[str(mode)]
343
344    code = f"\x1b[{mode}m"
345    if write:
346        get_terminal().write(code)
347
348    return code

377class MouseAction(Enum):
378    """An enumeration of all the polled mouse actions"""
379
380    LEFT_CLICK = "left_click"
381    """Start of a left button action sequence."""
382
383    LEFT_DRAG = "left_drag"
384    """Mouse moved while left button was held down."""
385
386    RIGHT_CLICK = "right_click"
387    """Start of a right button action sequence."""
388
389    RIGHT_DRAG = "right_drag"
390    """Mouse moved while right button was held down."""
391
392    SCROLL_UP = "scroll_up"
393    """Mouse wheel or touchpad scroll upwards."""
394
395    SCROLL_DOWN = "scroll_down"
396    """Mouse wheel or touchpad scroll downwards."""
397
398    SHIFT_SCROLL_UP = "shift_scroll_up"
399    """Mouse wheel or touchpad scroll upwards."""
400
401    SHIFT_SCROLL_DOWN = "shift_scroll_down"
402    """Mouse wheel or touchpad scroll downwards."""
403
404    HOVER = "hover"
405    """Mouse moved without clicking."""
406
407    # TODO: Support left & right mouse release separately, without breaking
408    #       current API.
409    RELEASE = "release"
410    """Mouse button released; end of any and all mouse action sequences."""

413@dataclass
414class MouseEvent:
415    """A class to represent events created by mouse actions.
416
417    Its first argument is a `MouseAction` describing what happened,
418    and its second argument is a `tuple[int, int]` describing where
419    it happened.
420
421    This class mostly exists for readability & typing reasons. It also
422    implements the iterable protocol, so you can use the unpacking syntax,
423    such as:
424
425    ```python3
426    action, position = MouseEvent(...)
427    ```
428    """
429
430    action: MouseAction
431    position: tuple[int, int]
432
433    def __post_init__(self) -> None:
434        """Initialize iteration counter"""
435
436        self._iter_index = 0
437
438    def __next__(self) -> MouseAction | tuple[int, int]:
439        """Get next iteration item"""
440
441        data = fields(self)
442
443        if self._iter_index >= len(data):
444            self._iter_index = 0
445            raise StopIteration
446
447        self._iter_index += 1
448        return getattr(self, data[self._iter_index - 1].name)
449
450    def __iter__(self) -> MouseEvent:
451        """Start iteration"""
452
453        return self
454
455    def is_scroll(self) -> bool:
456        """Returns True if event.action is one of the scrolling actions."""
457
458        return self.action in {MouseAction.SCROLL_DOWN, MouseAction.SCROLL_UP}
459
460    def is_primary(self) -> bool:
461        """Returns True if event.action is one of the primary (left-button) actions."""
462
463        return self.action in {MouseAction.LEFT_CLICK, MouseAction.LEFT_DRAG}
464
465    def is_secondary(self) -> bool:
466        """Returns True if event.action is one of the secondary (secondary-button) actions."""
467
468        return self.action in {MouseAction.RIGHT_CLICK, MouseAction.RIGHT_DRAG}

471def report_mouse(
472    event: str, method: Optional[str] = "decimal_xterm", stop: bool = False
473) -> None:
474    """Starts reporting of mouse events.
475
476    You can specify multiple events to report on.
477
478    Args:
479        event: The type of event to report on. See below for options.
480        method: The method of reporting to use. See below for options.
481        stop: If set to True, the stopping code is written to stdout.
482
483    Raises:
484        NotImplementedError: The given event is not supported.
485
486    Note:
487        If you need this functionality, you're probably better off using the wrapper
488        `pytermgui.context_managers.mouse_handler`, which allows listening on multiple
489        events, gives a translator method and handles exceptions.
490
491    Possible events:
492        - **press**: Report when the mouse is clicked, left or right button.
493        - **highlight**: Report highlighting.
494        - **press_hold**: Report with a left or right click, as well as both
495            left & right drag and release.
496        - **all**: Report every event, even hover.
497
498    Methods:
499        - **None**: Non-decimal xterm method. Limited in coordinates.
500        - **decimal_xterm**: The default setting. Most universally supported.
501        - **decimal_urxvt**: Older, less compatible, but useful on some systems.
502        - **decimal_utf8**:  Apparently not too stable.
503
504    More information <a href='https://stackoverflow.com/a/5970472'>here</a>.
505    """
506
507    terminal = get_terminal()
508
509    if event == "press":
510        terminal.write("\x1b[?1000")
511
512    elif event == "highlight":
513        terminal.write("\x1b[?1001")
514
515    elif event == "press_hold":
516        terminal.write("\x1b[?1002")
517
518    elif event == "all":
519        terminal.write("\x1b[?1003")
520
521    else:
522        raise NotImplementedError(f"Mouse report event {event!r} is not supported!")
523
524    terminal.write("l" if stop else "h")
525
526    if method == "decimal_utf8":
527        terminal.write("\x1b[?1005")
528
529    elif method == "decimal_xterm":
530        terminal.write("\x1b[?1006")
531
532    elif method == "decimal_urxvt":
533        terminal.write("\x1b[?1015")
534
535    elif method is None:
536        return
537
538    else:
539        raise NotImplementedError(f"Mouse report method {method} is not supported!")
540
541    terminal.write("l" if stop else "h", flush=True)

544def translate_mouse(code: str, method: str) -> list[MouseEvent | None] | None:
545    """Translates the output of produced by setting `report_mouse` into MouseEvents.
546
547    This method currently only supports `decimal_xterm` and `decimal_urxvt`.
548
549    Args:
550        code: The string of mouse code(s) to translate.
551        method: The reporting method to translate. One of `decimal_xterm`, `decimal_urxvt`.
552
553    Returns:
554        A list of optional mouse events obtained from the code argument. If the code was malformed,
555        and no codes could be determined None is returned.
556    """
557
558    if code == "\x1b":
559        return None
560
561    mouse_codes = {
562        "decimal_xterm": {
563            "0M": MouseAction.LEFT_CLICK,
564            "0m": MouseAction.RELEASE,
565            "2M": MouseAction.RIGHT_CLICK,
566            "2m": MouseAction.RELEASE,
567            "32": MouseAction.LEFT_DRAG,
568            "34": MouseAction.RIGHT_DRAG,
569            "35": MouseAction.HOVER,
570            "64": MouseAction.SCROLL_UP,
571            "65": MouseAction.SCROLL_DOWN,
572            "68": MouseAction.SHIFT_SCROLL_UP,
573            "69": MouseAction.SHIFT_SCROLL_DOWN,
574        },
575        "decimal_urxvt": {
576            "32": MouseAction.LEFT_CLICK,
577            "34": MouseAction.RIGHT_CLICK,
578            "35": MouseAction.RELEASE,
579            "64": MouseAction.LEFT_DRAG,
580            "66": MouseAction.RIGHT_DRAG,
581            "96": MouseAction.SCROLL_UP,
582            "97": MouseAction.SCROLL_DOWN,
583        },
584    }
585
586    mapping = mouse_codes[method]
587    pattern: Pattern = RE_MOUSE[method]
588
589    events: list[MouseEvent | None] = []
590
591    for sequence in code.split("\x1b"):
592        if len(sequence) == 0:
593            continue
594
595        matches = list(pattern.finditer(sequence))
596        if len(matches) == 0:
597            return None
598
599        for match in matches:
600            identifier, *pos, release_code = match.groups()
601
602            # decimal_xterm uses the last character's
603            # capitalization to signify press/release state
604            if len(release_code) > 0 and identifier in ["0", "2"]:
605                identifier += release_code
606
607            if identifier in mapping:
608                action = mapping[identifier]
609                assert isinstance(action, MouseAction)
610
611                events.append(MouseEvent(action, (int(pos[0]), int(pos[1]))))
612                continue
613
614            events.append(None)
615
616    return events

545def translate_mouse(code: str, method: str) -> list[MouseEvent | None] | None:
546    """Translates the output of produced by setting `report_mouse` into MouseEvents.
547
548    This method currently only supports `decimal_xterm` and `decimal_urxvt`.
549
550    Args:
551        code: The string of mouse code(s) to translate.
552        method: The reporting method to translate. One of `decimal_xterm`, `decimal_urxvt`.
553
554    Returns:
555        A list of optional mouse events obtained from the code argument. If the code was malformed,
556        and no codes could be determined None is returned.
557    """
558
559    if code == "\x1b":
560        return None
561
562    mouse_codes = {
563        "decimal_xterm": {
564            "0M": MouseAction.LEFT_CLICK,
565            "0m": MouseAction.RELEASE,
566            "2M": MouseAction.RIGHT_CLICK,
567            "2m": MouseAction.RELEASE,
568            "32": MouseAction.LEFT_DRAG,
569            "34": MouseAction.RIGHT_DRAG,
570            "35": MouseAction.HOVER,
571            "64": MouseAction.SCROLL_UP,
572            "65": MouseAction.SCROLL_DOWN,
573            "68": MouseAction.SHIFT_SCROLL_UP,
574            "69": MouseAction.SHIFT_SCROLL_DOWN,
575        },
576        "decimal_urxvt": {
577            "32": MouseAction.LEFT_CLICK,
578            "34": MouseAction.RIGHT_CLICK,
579            "35": MouseAction.RELEASE,
580            "64": MouseAction.LEFT_DRAG,
581            "66": MouseAction.RIGHT_DRAG,
582            "96": MouseAction.SCROLL_UP,
583            "97": MouseAction.SCROLL_DOWN,
584        },
585    }
586
587    mapping = mouse_codes[method]
588    pattern: Pattern = RE_MOUSE[method]
589
590    events: list[MouseEvent | None] = []
591
592    for sequence in code.split("\x1b"):
593        if len(sequence) == 0:
594            continue
595
596        matches = list(pattern.finditer(sequence))
597        if len(matches) == 0:
598            return None
599
600        for match in matches:
601            identifier, *pos, release_code = match.groups()
602
603            # decimal_xterm uses the last character's
604            # capitalization to signify press/release state
605            if len(release_code) > 0 and identifier in ["0", "2"]:
606                identifier += release_code
607
608            if identifier in mapping:
609                action = mapping[identifier]
610                assert isinstance(action, MouseAction)
611
612                events.append(MouseEvent(action, (int(pos[0]), int(pos[1]))))
613                continue
614
615            events.append(None)
616
617    return events
