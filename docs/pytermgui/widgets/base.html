310    def get_change(self) -> WidgetChange | None:
311        """Determines whether widget lines changed since the last call to this function."""
312
313        lines = self.get_lines()
314
315        if self._previous_state is None:
316            self._previous_state = (self.width, self.height), lines
317            return WidgetChange.LINES
318
319        lines = self.get_lines()
320        (old_width, old_height), old_lines = self._previous_state
321
322        self._previous_state = (self.width, self.height), lines
323
324        if old_width != self.width and old_height != self.height:
325            return WidgetChange.SIZE
326
327        if old_width != self.width:
328            return WidgetChange.WIDTH
329
330        if old_height != self.height:
331            return WidgetChange.HEIGHT
332
333        if old_lines != lines:
334            return WidgetChange.LINES
335
336        return None

358    def handle_mouse(self, event: MouseEvent) -> bool:
359        """Tries to call the most specific mouse handler function available.
360
361        This function looks for a set of mouse action handlers. Each handler follows
362        the format
363
364            on_{event_name}
365
366        For example, the handler triggered on MouseAction.LEFT_CLICK would be
367        `on_left_click`. If no handler is found nothing is done.
368
369        You can also define more general handlers, for example to group left & right
370        clicks you can use `on_click`, and to catch both up and down scroll you can use
371        `on_scroll`. General handlers are only used if they are the most specific ones,
372        i.e. there is no "specific" handler.
373
374        Args:
375            event: The event to handle.
376
377        Returns:
378            Whether the parent of this widget should treat it as one to "stick" events
379            to, e.g. to keep sending mouse events to it. One can "unstick" a widget by
380            returning False in the handler.
381        """
382
383        def _get_names(action: MouseAction) -> tuple[str, ...]:
384            if action.value in ["hover", "release"]:
385                return (action.value,)
386
387            parts = action.value.split("_")
388
389            # left click & right click
390            if parts[0] in ["left", "right"]:
391                return (action.value, parts[1])
392
393            if parts[0] == "shift":
394                return (action.value, f"shift_{parts[1]}", parts[1])
395
396            # scroll up & down
397            return (action.value, parts[0])
398
399        possible_names = _get_names(event.action)
400        for name in possible_names:
401            if hasattr(self, f"on_{name}"):
402                handle = getattr(self, f"on_{name}")
403
404                return handle(event)
405
406        return False

422    def serialize(self) -> dict[str, Any]:
423        """Serializes a widget.
424
425        The fields looked at are defined `Widget.serialized`. Note that
426        this method is not very commonly used at the moment, so it might
427        not have full functionality in non-nuclear widgets.
428
429        Returns:
430            Dictionary of widget attributes. The dictionary will always
431            have a `type` field. Any styles are converted into markup
432            strings during serialization, so they can be loaded again in
433            their original form.
434
435            Example return:
436            ```
437                {
438                    "type": "Label",
439                    "value": "[210 bold]I am a title",
440                    "parent_align": 0,
441                    ...
442                }
443            ```
444        """
445
446        fields = self._serialized_fields
447
448        out: dict[str, Any] = {"type": type(self).__name__}
449        for key in fields:
450            # Detect styled values
451            if key.startswith("*"):
452                style = True
453                key = key[1:]
454            else:
455                style = False
456
457            value = getattr(self, key)
458
459            # Convert styled value into markup
460            if style:
461                style_call = self._get_style(key)
462                if isinstance(value, list):
463                    out[key] = [get_markup(style_call(char)) for char in value]
464                else:
465                    out[key] = get_markup(style_call(value))
466
467                continue
468
469            out[key] = value
470
471        # The chars need to be handled separately
472        out["chars"] = {}
473        for key, value in self.chars.items():
474            style_call = self._get_style(key)
475
476            if isinstance(value, list):
477                out["chars"][key] = [get_markup(style_call(char)) for char in value]
478            else:
479                out["chars"][key] = get_markup(style_call(value))
480
481        return out

551    def bind(
552        self, key: str, action: BoundCallback, description: Optional[str] = None
553    ) -> None:
554        """Binds an action to a keypress.
555
556        This function is only called by implementations above this layer. To use this
557        functionality use `pytermgui.window_manager.WindowManager`, or write your own
558        custom layer.
559
560        Special keys:
561        - keys.ANY_KEY: Any and all keypresses execute this binding.
562        - keys.MouseAction: Any and all mouse inputs execute this binding.
563
564        Args:
565            key: The key that the action will be bound to.
566            action: The action executed when the key is pressed.
567            description: An optional description for this binding. It is not really
568                used anywhere, but you can provide a helper menu and display them.
569
570        Raises:
571            TypeError: This widget is not bindable, i.e. widget.is_bindable == False.
572        """
573
574        if not self.is_bindable:
575            raise TypeError(f"Widget of type {type(self)} does not accept bindings.")
576
577        if description is None:
578            description = f"Binding of {key} to {action}"
579
580        self._bindings[key] = (action, description)

587    def execute_binding(self, key: Any, ignore_any: bool = False) -> bool:
588        """Executes a binding belonging to key, when present.
589
590        Use this method inside custom widget `handle_keys` methods, or to run a callback
591        without its corresponding key having been pressed.
592
593        Args:
594            key: Usually a string, indexing into the `_bindings` dictionary. These are the
595                same strings as defined in `Widget.bind`.
596            ignore_any: If set, `keys.ANY_KEY` bindings will not be executed.
597
598        Returns:
599            True if the binding was found, False otherwise. Bindings will always be
600                executed if they are found.
601        """
602
603        # Execute special binding
604        if not ignore_any and keys.ANY_KEY in self._bindings:
605            method, _ = self._bindings[keys.ANY_KEY]
606            method(self, key)
607
608        if key in self._bindings:
609            method, _ = self._bindings[key]
610            method(self, key)
611
612            return True
613
614        return False

639    def debug(self) -> str:
640        """Returns identifiable information about this widget.
641
642        This method is used to easily differentiate between widgets. By default, all widget's
643        __repr__ method is an alias to this. The signature of each widget is used to generate
644        the return value.
645
646        Returns:
647            A string almost exactly matching the line of code that could have defined the widget.
648
649            Example return:
650
651            ```
652            Container(Label(value="This is a label", padding=0),
653            Button(label="This is a button", padding=0), **attrs)
654            ```
655
656        """
657
658        constructor = "("
659        for name in signature(getattr(self, "__init__")).parameters:
660            current = ""
661            if name == "attrs":
662                current += "**attrs"
663                continue
664
665            if len(constructor) > 1:
666                current += ", "
667
668            current += name
669
670            attr = getattr(self, name, None)
671            if attr is None:
672                continue
673
674            current += "="
675
676            if isinstance(attr, str):
677                current += f'"{attr}"'
678            else:
679                current += str(attr)
680
681            constructor += current
682
683        constructor += ")"
684
685        return type(self).__name__ + constructor

688class Label(Widget):
689    """A Widget to display a string
690
691    By default, this widget uses `pytermgui.widgets.styles.MARKUP`. This
692    allows it to house markup text that is parsed before display, such as:
693
694    ```python3
695    import pytermgui as ptg
696
697    with ptg.alt_buffer():
698        root = ptg.Container(
699            ptg.Label("[italic 141 bold]This is some [green]fancy [white inverse]text!")
700        )
701        root.print()
702        ptg.getch()
703    ```
704
705    <p style="text-align: center">
706     <img
707      src="https://github.com/bczsalba/pytermgui/blob/master/assets/docs/widgets/label.png?raw=true"
708      width=100%>
709    </p>
710    """
711
712    serialized = Widget.serialized + ["*value", "align", "padding"]
713    styles = w_styles.StyleManager(value="")
714
715    def __init__(
716        self,
717        value: str = "",
718        style: str | w_styles.StyleValue = "",
719        padding: int = 0,
720        non_first_padding: int = 0,
721        **attrs: Any,
722    ) -> None:
723        """Initializes a Label.
724
725        Args:
726            value: The value of this string. Using the default value style
727                (`pytermgui.widgets.styles.MARKUP`),
728            style: A pre-set value for self.styles.value.
729            padding: The number of space (" ") characters to prepend to every line after
730                line breaking.
731            non_first_padding: The number of space characters to prepend to every
732                non-first line of `get_lines`. This is applied on top of `padding`.
733        """
734
735        super().__init__(**attrs)
736
737        self.value = value
738        self.padding = padding
739        self.non_first_padding = non_first_padding
740        self.width = real_length(value) + self.padding
741
742        if style != "":
743            self.styles.value = style
744
745    def get_lines(self) -> list[str]:
746        """Get lines representing this Label, breaking lines as necessary"""
747
748        lines = []
749        limit = self.width - self.padding
750        broken = break_line(
751            self.styles.value(self.value),
752            limit=limit,
753            non_first_limit=limit - self.non_first_padding,
754        )
755
756        for i, line in enumerate(broken):
757            if i == 0:
758                lines.append(self.padding * " " + line)
759                continue
760
761            lines.append(self.padding * " " + self.non_first_padding * " " + line)
762
763        return lines or [""]
