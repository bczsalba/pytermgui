pytermgui.inspector

This module provides introspection utilities.

The inspect method can be used to create an Inspector widget, which can then be used to see what is happening inside any python object. This method is usually preferred for instantiating an Inspector, as it sets up overwriteable default arguments passed to the new widget.

These defaults are meant to hide the non-important information when they are not needed, in order to allow the least amount of code for the most usability. For example, by default, when passed a class, inspect will clip the docstrings to their first lines, but show all methods. When an class' method is given it will hide show the full docstring, and also use the method's fully qualified name.

 55class ObjectType(Enum):
 56    """All types an object can be."""
 57
 58    LIVE = _auto()
 59    """An instance that does not fit the other types."""
 60
 61    CLASS = _auto()
 62    """A class object."""
 63
 64    MODULE = _auto()
 65    """A module object."""
 66
 67    BUILTIN = _auto()
 68    """Some sort of a builtin object.
 69
 70    As builtins are often implemented in C, a lot of the standard python APIs
 71    won't work on them, so we need to treat them separately."""
 72
 73    FUNCTION = _auto()
 74    """A callable object, that is not a class."""

 77def _is_builtin(target: object) -> bool:
 78    """Determines if the given target is a builtin."""
 79
 80    try:
 81        signature(target)  # type: ignore
 82        return False
 83
 84    except (ValueError, TypeError):
 85        return True

 88def _determine_type(target: object) -> ObjectType:
 89    """Determines the type of an object."""
 90
 91    if ismodule(target):
 92        return ObjectType.MODULE
 93
 94    if _is_builtin(target):
 95        return ObjectType.BUILTIN
 96
 97    if isclass(target):
 98        return ObjectType.CLASS
 99
100    if isfunction(target) or callable(target):
101        return ObjectType.FUNCTION
102
103    return ObjectType.LIVE

106def _is_type_alias(obj: object) -> bool:
107    """Determines whether the given object is (likely) a type alias."""
108
109    return get_origin(obj) is not None

121def inspect(target: object, **inspector_args: Any) -> Inspector:
122    """Inspects an object.
123
124    Args:
125        target: The object to inspect.
126        **inspector_args: See `Inspector.__init__`.
127    """
128
129    def _conditionally_overwrite_kwarg(**kwargs) -> None:
130        for key, value in kwargs.items():
131            if inspector_args.get(key) is None:
132                inspector_args[key] = value
133
134    if ismodule(target):
135        _conditionally_overwrite_kwarg(
136            show_dunder=False,
137            show_private=False,
138            show_full_doc=False,
139            show_methods=True,
140            show_qualname=False,
141        )
142
143    elif isclass(target):
144        _conditionally_overwrite_kwarg(
145            show_dunder=False,
146            show_private=False,
147            show_full_doc=True,
148            show_methods=True,
149            show_qualname=False,
150        )
151
152    elif callable(target) or isbuiltin(target):
153        _conditionally_overwrite_kwarg(
154            show_dunder=False,
155            show_private=False,
156            show_full_doc=True,
157            show_methods=False,
158            show_qualname=True,
159        )
160
161    else:
162        _conditionally_overwrite_kwarg(
163            show_dunder=False,
164            show_private=False,
165            show_full_doc=True,
166            show_methods=True,
167            show_qualname=False,
168        )
169
170    inspector = Inspector(**inspector_args).inspect(target)
171
172    return inspector

175class Inspector(Container):
176    """A widget to inspect any Python object."""
177
178    def __init__(  # pylint: disable=too-many-arguments
179        self,
180        target: object = None,
181        show_private: bool = False,
182        show_dunder: bool = False,
183        show_methods: bool = False,
184        show_full_doc: bool = False,
185        show_qualname: bool = True,
186        show_header: bool = True,
187        **attrs: Any,
188    ):
189        """Initializes an inspector.
190
191        Note that most of the time, using `inspect` to do this is going to be more
192        useful.
193
194        Some styles of the inspector can be changed using the `code.name`,
195        `code.file` and `code.keyword` markup aliases. The rest of the
196        highlighting is done using `pprint`, with all of its respective colors.
197
198        Args:
199            show_private: Whether `_private` attributes should be shown.
200            show_dunder: Whether `__dunder__` attributes should be shown.
201            show_methods: Whether methods should be shown when encountering a class.
202            show_full_doc: If not set, docstrings are cut to only include their first
203                line.
204            show_qualname: Show fully-qualified name, e.g. `module.submodule.name`
205                instead of `name`.
206            show_header: If not set, the header containing the path to the object and
207                its qualname will not be added.
208        """
209
210        if "box" not in attrs:
211            attrs["box"] = "EMPTY"
212
213        super().__init__(**attrs)
214
215        self.width = self.terminal.width
216
217        self.show_private = show_private
218        self.show_dunder = show_dunder
219        self.show_methods = show_methods
220        self.show_full_doc = show_full_doc
221        self.show_qualname = show_qualname
222        self.show_header = show_header
223
224        # TODO: Fix attr-showing
225        self.show_attrs = False
226
227        self.target: object
228        if target is not None:
229            self.inspect(target)
230            self.target = target

232    def _get_header(self) -> Container:
233        """Creates a header containing the name and location of the object."""
234
235        header = Container(box="SINGLE")
236
237        line = "[code.name]"
238        if self.target_type is ObjectType.MODULE:
239            line += self.target.__name__  # type: ignore
240
241        else:
242            cls = (
243                self.target
244                if isclass(self.target) or isfunction(self.target)
245                else self.target.__class__
246            )
247            line += cls.__module__ + "." + cls.__qualname__  # type: ignore
248
249        header += line
250
251        try:
252            file = getfile(self.target)  # type: ignore
253        except TypeError:
254            return header
255
256        header += f"Located in [code.file ~file://{file}]{file}[/]"
257
258        return header
259
260    def _get_definition(self) -> Label:
261        """Returns the definition str of self.target."""
262
263        target = self.target
264
265        if self.show_qualname:
266            name = getattr(target, "__qualname__", type(target).__name__)
267        else:
268            name = getattr(target, "__name__", type(target).__name__)
269
270        if self.target_type == ObjectType.LIVE:
271            target = type(target)
272
273        otype = _determine_type(target)
274
275        keyword = ""
276        if otype == ObjectType.CLASS:
277            keyword = "class "
278
279        elif otype == ObjectType.FUNCTION:
280            keyword = "def "
281
282        try:
283            assert callable(target)
284            definition = self.highlight(keyword + name + str(signature(target)) + ":")
285
286        except (TypeError, ValueError, AssertionError):
287            definition = self.highlight(keyword + name + "(...)")
288
289        return Label(definition, parent_align=0, non_first_padding=4)
290
291    def _get_docs(self, padding: int) -> Label:
292        """Returns a list of Labels of the object's documentation."""
293
294        default = Label("...", style="102")
295        if self.target.__doc__ is None:
296            return default
297
298        doc = getdoc(self.target)
299
300        if doc is None:
301            return default
302
303        lines = doc.splitlines()
304        if not self.show_full_doc and len(lines) > 0:
305            lines = [lines[0]]
306
307        trimmed = "\n".join(lines)
308
309        return Label(
310            trimmed.replace("[", r"\["),
311            style="102",
312            parent_align=0,
313            padding=padding,
314        )
315
316    def _get_keys(self) -> list[str]:
317        """Gets all inspectable keys of an object.
318
319        It first checks for an `__all__` attribute, and substitutes `dir` if not found.
320        Then, if there are too many keys and the given target is a module it tries to
321        list all of the present submodules.
322        """
323
324        keys = getattr(self.target, "__all__", dir(self.target))
325
326        if not self.show_dunder:
327            keys = [key for key in keys if not key.startswith("__")]
328
329        if not self.show_private:
330            keys = [key for key in keys if not (key.startswith("_") and key[1] != "_")]
331
332        if not self.show_methods:
333            keys = [
334                key for key in keys if not callable(getattr(self.target, key, None))
335            ]
336
337        keys.sort(key=lambda item: callable(getattr(self.target, item, None)))
338
339        return keys

341    def _get_preview(self) -> Container:
342        """Gets a Container with self.target inside."""
343
344        preview = Container(static_width=self.width // 2, parent_align=0, box="SINGLE")
345
346        if isinstance(self.target, str) and RE_MARKUP.match(self.target) is not None:
347            preview += Label(prettify(self.target, parse=False), parent_align=0)
348            return preview
349
350        for line in prettify(self.target).splitlines():
351
352            if real_length(line) > preview.width - preview.sidelength:
353                preview.width = real_length(line) + preview.sidelength
354
355            preview += Label(tim.get_markup(line), parent_align=0)
356
357        preview.width = min(preview.width, self.terminal.width - preview.sidelength)
358        return preview

360    @staticmethod
361    def highlight(text: str) -> str:
362        """Applies highlighting to a given string.
363
364        This highlight includes keywords, builtin types and more.
365
366        Args:
367            text: The string to highlight.
368
369        Returns:
370            Unparsed markup.
371        """
372
373        def _split(text: str, chars: str = " ,:|()[]{}") -> list[tuple[str, str]]:
374            """Splits given text by the given chars.
375
376            Args:
377                text: The text to split.
378                chars: A string of characters we will split by.
379
380            Returns:
381                A tuple of (delimiter, word) tuples. Delimiter is one of the characters
382                of `chars`.
383            """
384
385            last_delim = ""
386            output = []
387            word = ""
388            for char in text:
389                if char in chars:
390                    output.append((last_delim, word))
391                    last_delim = char
392                    word = ""
393                    continue
394
395                word += char
396
397            output.append((last_delim, word))
398            return output
399
400        buff = ""
401        for (delim, word) in _split(text):
402            stripped = word.strip("'")
403            highlighted = highlight_python(stripped)
404
405            if highlighted != stripped:
406                buff += delim + stripped
407                continue
408
409            buff += delim + stripped
410
411        return highlight_python(buff)

413    def inspect(self, target: object) -> Inspector:
414        """Inspects a given object, and sets self.target to it.
415
416        Returns:
417            Self, with the new content based on the inspection.
418        """
419
420        self.target = target
421        self.target_type = _determine_type(target)
422
423        # Header
424        if self.show_header and self.box is not INDENTED_EMPTY_BOX:
425            self.lazy_add(self._get_header())
426
427        # Body
428        if self.target_type is not ObjectType.MODULE:
429            self.lazy_add(self._get_definition())
430
431        padding = 0 if self.target_type is ObjectType.MODULE else 4
432
433        self.lazy_add(self._get_docs(padding))
434
435        keys = self._get_keys()
436
437        for key in keys:
438            attr = getattr(target, key, None)
439
440            # Don't show type aliases
441            if _is_type_alias(attr):
442                continue
443
444            # Only show functions if they are not lambdas
445            if (isfunction(attr) or callable(attr)) and (
446                hasattr(attr, "__name__") and not attr.__name__ == "<lambda>"
447            ):
448                self.lazy_add(
449                    Inspector(
450                        box=INDENTED_EMPTY_BOX,
451                        show_dunder=self.show_dunder,
452                        show_private=self.show_private,
453                        show_full_doc=False,
454                        show_qualname=self.show_qualname,
455                    ).inspect(attr)
456                )
457                continue
458
459            if not self.show_attrs:
460                continue
461
462            for i, line in enumerate(prettify(attr, parse=False).splitlines()):
463                if i == 0:
464                    line = f"- {key}: {line}"
465
466                self.lazy_add(Label(line, parent_align=0))
467
468        # Footer
469        if self.target_type in [ObjectType.LIVE, ObjectType.BUILTIN]:
470            self.lazy_add(self._get_preview())
471
472        return self

474    def debug(self) -> str:
475        """Returns identifiable information used in repr."""
476
477        if self.terminal.is_interactive and not self.terminal.displayhook_installed:
478            return "\n".join(self.get_lines())
479
480        return Widget.debug(self)
