300    def resolution(self) -> tuple[int, int]:
301        """Returns the terminal's pixel based resolution.
302
303        Only evaluated on demand.
304        """
305
306        if self.isatty():
307            sys.stdout.write("\x1b[14t")
308            sys.stdout.flush()
309
310            # Some terminals may not respond to a pixel size query, so we send
311            # a timed-out getch call with a default response of 1280x720.
312            output = getch_timeout(0.1, default="\x1b[4;720;1280t")
313            match = RE_PIXEL_SIZE.match(output)

315            if match is not None:
316                return (int(match[2]), int(match[1]))
317
318        return (0, 0)

447    @contextmanager
448    def frame(self) -> Generator[StringIO, None, None]:
449        """Notifies the emulator of the inner content being a single frame.
450
451        See https://gist.github.com/christianparpart/d8a62cc1ab659194337d73e399004036!
452        """
453
454        buffer = StringIO()
455
456        try:
457            with self.no_record():
458                self.write("\x1b[?2026h")

473    def replay(self, recorder: Recorder) -> None:
474        """Replays a recording."""
475
476        last_time = 0.0
477        for data, delay in recorder.recording:
478            if last_time > 0.0:
479                time.sleep(delay - last_time)

481            self.write(data, flush=True)

484    def subscribe(self, event: int, callback: Callable[..., Any]) -> None:
485        """Subscribes a callback to be called when event occurs.

499    def write(
500        self,
501        data: str,
502        pos: tuple[int, int] | None = None,
503        flush: bool = False,
504        slice_too_long: bool = True,
505    ) -> None:
506        """Writes the given data to the terminal's stream.
507
508        Args:
509            data: The data to write.
510            pos: Terminal-character space position to write the data to, (x, y).
511            flush: If set, `flush` will be called on the stream after reading.
512            slice_too_long: If set, lines that are outside of the terminal will be
513                sliced to fit. Involves a sizable performance hit.
514        """
515
516        def _slice(line: str, maximum: int) -> str:
517            length = 0
518            sliced = ""
519            for char in line:
520                sliced += char
521                if char == "\x1b":
522                    continue
523
524                if (
525                    length > maximum
526                    and real_length(sliced) > maximum
527                    and not has_open_sequence(sliced)
528                ):
529                    break
530
531                length += 1
532
533            return sliced
534
535        if "\x1b[2J" in data:
536            self.clear_stream()
537
538        if pos is not None:
539            xpos, ypos = pos
540
541            if slice_too_long:
542                if not self.height + self.origin[1] + 1 > ypos >= 0:
543                    return
544
545                maximum = self.width - xpos + self.origin[0]
546
547                if xpos < self.origin[0]:
548                    xpos = self.origin[0]
549
550                sliced = _slice(data, maximum) if len(data) > maximum else data
551
552                data = f"\x1b[{ypos};{xpos}H{sliced}\x1b[0m"
553
554            else:
555                data = f"\x1b[{ypos};{xpos}H{data}"
556
557        self._stream.write(data)
558
559        if self._recorder is not None:
560            self._recorder.write(data)
561
562        if flush:
563            self._stream.flush()
564
565    def clear_stream(self) -> None:
566        """Clears (truncates) the terminal's stream."""
567
568        try:
569            self._stream.truncate(0)
570
571        except OSError as error:
572            if error.errno != errno.EINVAL and os.name != "nt":
573                raise
574
575        self._stream.write("\x1b[2J")
