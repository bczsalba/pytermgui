 95def _get_palette_color(color: Literal["10", "11"]) -> Color:
 96    """Gets either the foreground or background color of the current emulator.
 97
 98    Args:
 99        color: The value used for `Ps` in the query. See https://unix.stackexchange.com/a/172674.
100    """
101
102    defaults = {
103        "10": RGBColor.from_rgb((222, 222, 222)),
104        "11": RGBColor.from_rgb((20, 20, 20)),
105    }
106
107    if not terminal.isatty():
108        return defaults[color]
109
110    sys.stdout.write(f"\x1b]{color};?\007")
111    sys.stdout.flush()
112
113    reply = getch()
114
115    match = RE_PALETTE_REPLY.match(reply)
116    if match is None:
117        return defaults[color]
118
119    _, red, green, blue = match.groups()
120
121    rgb: list[int] = []
122    for part in (red, green, blue):
123        rgb.append(int(part[:2], base=16))
124
125    palette_color = RGBColor.from_rgb(tuple(rgb))  # type: ignore
126    palette_color.background = color == "11"
127
128    return palette_color

133class Color:  # pylint: disable=too-many-public-methods
134    """A terminal color.
135
136    Args:
137        value: The data contained within this color.
138        background: Whether this color will represent a color.
139
140    These colors are all formattable. There are currently 2 'spec' strings:
141    - f"{my_color:tim}" -> Returns self.markup
142    - f"{my_color:seq}" -> Returns self.sequence
143
144    They can thus be used in TIM strings:
145
146        >>> ptg.tim.parse("[{my_color:tim}]Hello")
147        '[<my_color.markup>]Hello'

150
151        >>> "{my_color:seq}Hello"
152        '<my_color.sequence>Hello'
153    """

301    @cached_property
302    def luminance(self) -> float:
303        """Returns this color's perceived luminance (brightness).
304
305        From https://stackoverflow.com/a/596243
306        """
307
308        def _linearize(color: float) -> float:
309            """Converts sRGB color to linear value."""
310
311            if color <= 0.04045:
312                return color / 12.92
313
314            return ((color + 0.055) / 1.055) ** 2.4
315
316        red, green, blue = float(self.rgb[0]), float(self.rgb[1]), float(self.rgb[2])
317
318        red /= 255
319        green /= 255
320        blue /= 255
321
322        red = _linearize(red)
323        blue = _linearize(blue)
324        green = _linearize(green)
325
326        return 0.2126 * red + 0.7152 * green + 0.0722 * blue

328    def hue_offset(self, offset: float) -> Color:
329        """Returns the color offset by the given hue."""
330
331        hue, lightness, saturation = colorsys.rgb_to_hls(
332            self.red / 256, self.green / 256, self.blue / 256
333        )
334
335        hue = (hue + offset) % 1
336
337        return Color.parse(
338            ";".join(
339                map(
340                    lambda n: str(int(256 * n)),

342                )

345            localize=False,

351

355        if self.luminance <= (216 / 24389):
356            brightness = self.luminance * (24389 / 27)

421    def blend(self, other: Color, alpha: float = 0.5, localize: bool = False) -> Color:
422        """Blends a color into another one.
423
424        Args:
425            other: The color to blend with.
426            alpha: How much the other color should influence the outcome.
427            localize: If set, the returned color will returned its localized version by running
428                `get_localized` on it before returning.
429
430        Returns:
431            A `Color` that is the result of the blending.
432        """

434        red1, green1, blue1 = self.rgb
435        red2, green2, blue2 = other.rgb
436
437        blended: Color = RGBColor.from_rgb(
438            (
439                int(red1 + (red2 - red1) * alpha),
440                int(green1 + (green2 - green1) * alpha),
441                int(blue1 + (blue2 - blue1) * alpha),
442            )
443        )
444
445        if localize:
446            blended = blended.get_localized()
447
448        return blended

521@dataclass(repr=False)
522class IndexedColor(Color):
523    """A color representing an index into the xterm-256 color palette."""

527    def __post_init__(self) -> None:
528        """Ensures data validity."""
529
530        if not self.value.isdigit():

534

536            raise ValueError(

541        """Yields a fancy looking string."""
542
543        yield f"<{type(self).__name__} value: {self.value}, preview: "

549    @classmethod
550    def from_rgb(cls, rgb: RGBTriplet) -> IndexedColor:
551        """Constructs an `IndexedColor` from the closest matching option."""
552
553        if rgb in _COLOR_MATCH_CACHE:
554            color = _COLOR_MATCH_CACHE[rgb]
555
556            assert isinstance(color, IndexedColor)
557            return color
558
559        if terminal.colorsystem == ColorSystem.STANDARD:
560            return StandardColor.from_rgb(rgb)
561
562        # Normalize the color values
563        red, green, blue = (x / 255 for x in rgb)
564
565        # Calculate the eight-bit color index
566        color_num = 16
567        color_num += 36 * round(red * 5.0)
568        color_num += 6 * round(green * 5.0)
569        color_num += round(blue * 5.0)
570
571        color = cls(str(color_num))
572        _COLOR_MATCH_CACHE[rgb] = color
573
574        return color

581
582        return "\x1b[" + ("48" if self.background else "38") + f";5;{index}m"

597class StandardColor(IndexedColor):
598    """A color in the xterm-16 palette."""
599
600    system = ColorSystem.STANDARD
601
602    @property
603    def name(self) -> str:
604        """Returns the markup-compatible name for this color."""
605
606        index = name = int(self.value)
607
608        # Normal colors
609        if 30 <= index <= 47:
610            name -= 30
611
612        elif 90 <= index <= 107:
613            name -= 82
614
615        return ("@" if self.background else "") + str(name)

617    @classmethod
618    def from_ansi(cls, code: str) -> StandardColor:
619        """Creates a standard color from the given ANSI code.
620
621        These codes have to be a digit ranging between 31 and 47.
622        """
623
624        if not code.isdigit():
625            raise ColorSyntaxError(
626                f"Standard color codes must be digits, not {code!r}."
627            )
628
629        code_int = int(code)
630
631        if not 30 <= code_int <= 47 and not 90 <= code_int <= 107:
632            raise ColorSyntaxError(
633                f"Standard color codes must be in the range ]30;47[ or ]90;107[, got {code_int!r}."
634            )
635
636        is_background = 40 <= code_int <= 47 or 100 <= code_int <= 107
637
638        if is_background:
639            code_int -= 10
640
641        return cls(str(code_int), background=is_background)
642
643    @classmethod
644    def from_rgb(cls, rgb: RGBTriplet) -> StandardColor:
645        """Creates a color with the closest-matching xterm index, based on rgb.
646
647        Args:
648            rgb: The target color.
649        """
650
651        if rgb in _COLOR_MATCH_CACHE:
652            color = _COLOR_MATCH_CACHE[rgb]
653
654            if color.system is ColorSystem.STANDARD:
655                assert isinstance(color, StandardColor)
656                return color
657
658        # Find the least-different color in the table
659        index = min(range(16), key=lambda i: _get_color_difference(rgb, COLOR_TABLE[i]))
660
661        if index > 7:
662            index += 82
663        else:
664            index += 30
665
666        color = cls(str(index))
667
668        _COLOR_MATCH_CACHE[rgb] = color
669
670        return color

678        if self.background:
679            index += 10
