 118    @property
 119    def selectables(self) -> list[tuple[Widget, int]]:
 120        """Gets all selectable widgets and their inner indices.
 121
 122        This is used in order to have a constant reference to all selectable indices within this
 123        widget.
 124
 125        Returns:
 126            A list of tuples containing a widget and an integer each. For each widget that is
 127            withing this one, it is added to this list as many times as it has selectables. Each
 128            of the integers correspond to a selectable_index within the widget.
 129
 130            For example, a Container with a Button, InputField and an inner Container containing
 131            3 selectables might return something like this:
 132
 133            ```
 134            [
 135                (Button(...), 0),
 136                (InputField(...), 0),
 137                (Container(...), 0),
 138                (Container(...), 1),
 139                (Container(...), 2),
 140            ]
 141            ```
 142        """
 143
 144        _selectables: list[tuple[Widget, int]] = []
 145        for widget in self._widgets:
 146            if not widget.is_selectable:
 147                continue
 148
 149            for i, (inner, _) in enumerate(widget.selectables):
 150                _selectables.append((inner, i))

 152        return _selectables

 317    def _add_widget(self, other: object, run_get_lines: bool = True) -> Widget:
 318        """Adds other to this widget.
 319
 320        Args:
 321            other: Any widget-like object.
 322            run_get_lines: Boolean controlling whether the self.get_lines is ran.
 323
 324        Returns:
 325            The added widget. This is useful when data conversion took place in this
 326            function, e.g. a string was converted to a Label.
 327        """

 329        if not isinstance(other, Widget):
 330            to_widget = Widget.from_data(other)
 331            if to_widget is None:
 332                raise ValueError(
 333                    f"Could not convert {other} of type {type(other)} to a Widget!"
 334                )

 336            other = to_widget

 338        # This is safe to do, as it would've raised an exception above already
 339        assert isinstance(other, Widget)

 343            other.set_recursive_depth(self.depth + 2)
 344        else:
 345            other.depth = self.depth + 1

 347        other.get_lines()
 348        other.parent = self
 349
 350        if run_get_lines:
 351            self.get_lines()

 355    def _get_aligners(
 356        self, widget: Widget, borders: tuple[str, str]
 357    ) -> tuple[Callable[[str], str], int]:
 358        """Gets an aligning method and position offset.
 359
 360        Args:
 361            widget: The widget to align.
 362            borders: The left and right borders to put the widget within.
 363
 364        Returns:
 365            A tuple of a method that, when called with a line, will return that line
 366            centered using the passed in widget's parent_align and width, as well as
 367            the horizontal offset resulting from the widget being aligned.
 368        """
 369
 370        left, right = self.styles.border(borders[0]), self.styles.border(borders[1])
 371        char = " "
 372
 373        fill = self.styles.fill
 374
 375        def _align_left(text: str) -> str:
 376            """Align line to the left"""
 377
 378            padding = self.width - real_length(left + right) - real_length(text)
 379            return left + text + fill(padding * char) + right
 380
 381        def _align_center(text: str) -> str:
 382            """Align line to the center"""
 383
 384            total = self.width - real_length(left + right) - real_length(text)
 385            padding, offset = divmod(total, 2)
 386            return (
 387                left
 388                + fill((padding + offset) * char)
 389                + text
 390                + fill(padding * char)
 391                + right
 392            )
 393
 394        def _align_right(text: str) -> str:
 395            """Align line to the right"""
 396
 397            padding = self.width - real_length(left + right) - real_length(text)
 398            return left + fill(padding * char) + text + right
 399
 400        if widget.parent_align == HorizontalAlignment.CENTER:
 401            total = self.width - real_length(left + right) - widget.width
 402            padding, offset = divmod(total, 2)
 403            return _align_center, real_length(left) + padding + offset
 404
 405        if widget.parent_align == HorizontalAlignment.RIGHT:
 406            return _align_right, self.width - real_length(left) - widget.width
 407
 408        # Default to left-aligned
 409        return _align_left, real_length(left)

 411    def _update_width(self, widget: Widget) -> None:
 412        """Updates the width of widget or self.
 413
 414        This method respects widget.size_policy.
 415
 416        Args:
 417            widget: The widget to update/base updates on.
 418
 419        Raises:
 420            ValueError: Widget has SizePolicy.RELATIVE, but relative_width is None.
 421            WidthExceededError: Widget and self both have static widths, and widget's
 422                is larger than what is available.
 423        """
 424
 425        available = self.width - self.sidelength
 426
 427        if widget.size_policy == SizePolicy.FILL:
 428            widget.width = available
 429            return
 430
 431        if widget.size_policy == SizePolicy.RELATIVE:
 432            if widget.relative_width is None:
 433                raise ValueError(f'Widget "{widget}"\'s relative width cannot be None.')
 434
 435            widget.width = int(widget.relative_width * available)
 436            return
 437
 438        if widget.width > available:
 439            if widget.size_policy == self.size_policy == SizePolicy.STATIC:
 440                raise WidthExceededError(
 441                    f"Widget {widget}'s static width of {widget.width}"
 442                    + f" exceeds its parent's available width {available}."
 443                    ""
 444                )
 445
 446            if widget.size_policy == SizePolicy.STATIC:
 447                self.width = widget.width + self.sidelength
 448
 449            else:
 450                widget.width = available

 452    def _apply_vertalign(
 453        self, lines: list[str], diff: int, padder: str
 454    ) -> tuple[int, list[str]]:
 455        """Insert padder line into lines diff times, depending on self.vertical_align.
 456
 457        Args:
 458            lines: The list of lines to align.
 459            diff: The available height.
 460            padder: The line to use to pad.
 461
 462        Returns:
 463            A tuple containing the vertical offset as well as the padded list of lines.
 464
 465        Raises:
 466            NotImplementedError: The given vertical alignment is not implemented.
 467        """
 468
 469        if self.vertical_align == VerticalAlignment.BOTTOM:
 470            for _ in range(diff):
 471                lines.insert(0, padder)
 472
 473            return diff, lines
 474
 475        if self.vertical_align == VerticalAlignment.TOP:
 476            for _ in range(diff):
 477                lines.append(padder)
 478
 479            return 0, lines
 480
 481        if self.vertical_align == VerticalAlignment.CENTER:
 482            top, extra = divmod(diff, 2)
 483            bottom = top + extra
 484
 485            for _ in range(top):
 486                lines.insert(0, padder)
 487
 488            for _ in range(bottom):
 489                lines.append(padder)
 490
 491            return top, lines
 492
 493        raise NotImplementedError(
 494            f"Vertical alignment {self.vertical_align} is not implemented for {type(self)}."
 495        )

 508    def get_lines(self) -> list[str]:
 509        """Gets all lines by spacing out inner widgets.
 510
 511        This method reflects & applies both width settings, as well as
 512        the `parent_align` field.
 513
 514        Returns:
 515            A list of all lines that represent this Container.
 516        """
 517
 518        def _get_border(left: str, char: str, right: str) -> str:
 519            """Gets a top or bottom border.
 520
 521            Args:
 522                left: Left corner character.
 523                char: Border character filling between left & right.
 524                right: Right corner character.
 525
 526            Returns:
 527                The border line.
 528            """
 529
 530            offset = real_length(strip_markup(left + right))
 531            return (
 532                self.styles.corner(left)
 533                + self.styles.border(char * (self.width - offset))
 534                + self.styles.corner(right)
 535            )
 536
 537        lines: list[str] = []
 538
 539        borders = self._get_char("border")
 540        corners = self._get_char("corner")
 541
 542        has_top_bottom = (real_length(borders[1]) > 0, real_length(borders[3]) > 0)
 543
 544        align, offset = self._get_aligners(self, (borders[0], borders[2]))
 545
 546        overflow = self.overflow
 547
 548        for widget in self._widgets:
 549            align, offset = self._get_aligners(widget, (borders[0], borders[2]))
 550
 551            self._update_width(widget)
 552
 553            widget.pos = (
 554                self.pos[0] + offset,
 555                self.pos[1] + len(lines) + (1 if has_top_bottom[0] else 0),
 556            )
 557
 558            widget_lines: list[str] = []
 559            for line in widget.get_lines():
 560                if len(lines) + len(widget_lines) >= self.height - sum(has_top_bottom):
 561                    if overflow is Overflow.HIDE:
 562                        break
 563
 564                    if overflow == Overflow.AUTO:
 565                        overflow = Overflow.SCROLL
 566
 567                widget_lines.append(align(line))
 568
 569            lines.extend(widget_lines)
 570
 571        if overflow == Overflow.SCROLL:
 572            self._max_scroll = len(lines) - self.height + sum(has_top_bottom)
 573            height = self.height - sum(has_top_bottom)
 574
 575            self._scroll_offset = max(0, min(self._scroll_offset, len(lines) - height))
 576            lines = lines[self._scroll_offset : self._scroll_offset + height]
 577
 578        elif overflow == Overflow.RESIZE:
 579            self.height = len(lines) + sum(has_top_bottom)
 580
 581        vertical_offset, lines = self._apply_vertalign(
 582            lines, self.height - len(lines) - sum(has_top_bottom), align("")
 583        )
 584
 585        for widget in self._widgets:
 586            widget.pos = (widget.pos[0], widget.pos[1] + vertical_offset)
 587
 588            if widget.is_selectable:
 589                # This buffer will be out of position, so we must clear it.
 590                widget.positioned_line_buffer = []
 591                widget.get_lines()
 592
 593            self.positioned_line_buffer.extend(widget.positioned_line_buffer)
 594
 595            widget.positioned_line_buffer = []
 596
 597        if has_top_bottom[0]:
 598            lines.insert(0, _get_border(corners[0], borders[1], corners[1]))
 599
 600        if has_top_bottom[1]:
 601            lines.append(_get_border(corners[3], borders[3], corners[2]))
 602
 603        self.height = len(lines)
 604        return lines

 738    def handle_mouse(self, event: MouseEvent) -> bool:
 739        """Handles mouse events.
 740
 741        This, like all mouse handlers should, calls super()'s implementation first,
 742        to allow usage of `on_{event}`-type callbacks. After that, it tries to find
 743        a target widget within itself to handle the event.
 744
 745        Each handler will return a boolean. This boolean is then used to figure out
 746        whether the targeted widget should be "sticky", i.e. a slider. Returning
 747        True will set that widget as the current mouse target, and all mouse events will
 748        be sent to it as long as it returns True.
 749
 750        Args:
 751            event: The event to handle.
 752
 753        Returns:
 754            Whether the parent of this widget should treat it as one to "stick" events
 755            to, e.g. to keep sending mouse events to it. One can "unstick" a widget by
 756            returning False in the handler.
 757        """
 758
 759        def _handle_scrolling() -> bool:
 760            """Scrolls the container."""
 761
 762            if self.overflow != Overflow.SCROLL:
 763                return False
 764
 765            if event.action is MouseAction.SCROLL_UP:
 766                return self.scroll(-1)

 856        self, key: str

 858        """Handles a keypress, returns its success.
 859
 860        Args:
 861            key: A key str.
 862
 863        Returns:

 865        """

 868            """Determine if a key is in the navigation sets"""

 870            return key in self.keys["next"] | self.keys["previous"]
 871
 872        if self.selected is not None and self.selected.handle_key(key):
 873            return True
 874
 875        scroll_actions = {
 876            **{key: 1 for key in self.keys["scroll_down"]},
 877            **{key: -1 for key in self.keys["scroll_up"]},
 878        }

 882                if isinstance(widget, Container) and self.selected in widget:
 883                    widget.handle_key(key)

 891                self.select(0)

 904                self.select(self.selected_index - 1)
 905                handled = True
 906
 907            elif key in self.keys["next"]:
 908                # Stop selection at last element, return as unhandled

 911                    return False
 912
 913                self.select(new)
 914                handled = True
 915
 916            if handled:
 917                return True
 918

 921                self.select(0)
 922

 925                return True
 926
 927        for widget in self._widgets:
 928            if widget.execute_binding(key):
 929                return True
 930
 931        return False

739    def handle_mouse(self, event: MouseEvent) -> bool:

743        to allow usage of `on_{event}`-type callbacks. After that, it tries to find
744        a target widget within itself to handle the event.
745
746        Each handler will return a boolean. This boolean is then used to figure out
747        whether the targeted widget should be "sticky", i.e. a slider. Returning
748        True will set that widget as the current mouse target, and all mouse events will
749        be sent to it as long as it returns True.
750

752            event: The event to handle.
753

755            Whether the parent of this widget should treat it as one to "stick" events
756            to, e.g. to keep sending mouse events to it. One can "unstick" a widget by
757            returning False in the handler.
758        """
759
760        def _handle_scrolling() -> bool:
761            """Scrolls the container."""
762
763            if self.overflow != Overflow.SCROLL:
764                return False
765
766            if event.action is MouseAction.SCROLL_UP:
767                return self.scroll(-1)
768
769            if event.action is MouseAction.SCROLL_DOWN:
770                return self.scroll(1)
771
772            return False
773
774        if super().handle_mouse(event):
775            return True

777        if event.action is MouseAction.RELEASE and self._mouse_target is not None:
778            return self._mouse_target.handle_mouse(event)
779
780        if (
781            self._mouse_target is not None
782            and (
783                event.action.value.endswith("drag")
784                or event.action.value.startswith("scroll")
785            )
786            and self._mouse_target.handle_mouse(event)
787        ):
788            return True
789
790        release = MouseEvent(MouseAction.RELEASE, event.position)
791
792        selectables_index = 0
793        event.position = (event.position[0], event.position[1] + self._scroll_offset)
794
795        handled = False
796        for widget in self._widgets:
797            if (
798                widget.pos[1] - self.pos[1] - self._scroll_offset
799                > self.content_dimensions[1]
800            ):
801                break
802
803            if widget.contains(event.position):
804                handled = widget.handle_mouse(event)
805                selectables_index += widget.selected_index or 0
806
807                # TODO: This really should be customizable somehow.
808                if event.action is MouseAction.LEFT_CLICK:
809                    if handled and selectables_index < len(self.selectables):
810                        self.select(selectables_index)
811
812                if self._mouse_target is not None and self._mouse_target is not widget:
813                    self._mouse_target.handle_mouse(release)
814
815                self._mouse_target = widget
816
817                break

824        return handled

856    def handle_key(  # pylint: disable=too-many-return-statements, too-many-branches
857        self, key: str

872

877            **{key: 1 for key in self.keys["scroll_down"]},
878            **{key: -1 for key in self.keys["scroll_up"]},

892                self.select(0)
893                return True

896
897            assert isinstance(self.selected_index, int)

899            if key in self.keys["previous"]:

901                # upwards.

904
905                self.select(self.selected_index - 1)

907

926                return True
